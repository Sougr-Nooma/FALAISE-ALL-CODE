import { useState, useRef } from "react";
export default function ReclamationSection() {
  const [form, setForm] = useState({
    nom: "",
    telephone: "",
    email: "",
    source: "",
    photo: null,
    message: "",
  });
  const [errors, setErrors] = useState({});
  const [sent, setSent] = useState(false);
  const fileInput = useRef(null);

  const handleChange = (e) => {
    const { name, value, files } = e.target;
    setForm((f) => ({
      ...f,
      [name]: name === "photo" ? files[0] : value,
    }));
    if (errors[name]) setErrors((err) => ({ ...err, [name]: "" }));
  };

  const validate = () => {
    const err = {};
    if (!form.message.trim()) err.message = "Veuillez décrire votre problème";
    if (!form.source) err.source = "Sélectionnez une source";
    return err;
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    const err = validate();
    if (Object.keys(err).length) {
      setErrors(err);
      return;
    }
    console.log("Réclamation :", form);
    setSent(true);
  };

  if (sent)
    return (
      <div className="max-w-4xl mx-auto bg-green-50 text-green-800 rounded-lg p-6 text-center">
        <h3 className="text-xl font-semibold">Merci pour votre retour !</h3>
        <p className="mt-2">Nous traiterons votre demande dans les plus brefs délais.</p>
        <button
          onClick={() => setSent(false)}
          className="mt-4 bg-primary text-white px-4 py-2 rounded-lg hover:bg-orange-600 transition"
        >
          Nouveau message
        </button>
      </div>
    );

  return (
    <section id="reclamation" className="py-24 bg-gray-50">
      <div className="max-w-4xl mx-auto px-6">
        
        <h2 className="text-4xl font-bold text-secondary text-center mb-8">
                Votre Avis Nous Intérresse
         </h2>
        <form
          onSubmit={handleSubmit}
          className="bg-white rounded-2xl shadow p-8 grid md:grid-cols-2 gap-6"
        >
            
         
          {/* Nom */}
          <div>
            <label className="block text-sm font-medium mb-1">Nom (optionnel)</label>
            <input
              name="nom"
              value={form.nom}
              onChange={handleChange}
              className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-primary"
              placeholder="Jean Dupont"
            />
          </div>

          {/* Téléphone */}
          <div>
            <label className="block text-sm font-medium mb-1">
              Téléphone (optionnel mais utile)
            </label>
            <input
              name="telephone"
              type="tel"
              value={form.telephone}
              onChange={handleChange}
              className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-primary"
              placeholder="+216 06 000 000"
            />
          </div>

          {/* Email */}
          <div className="md:col-span-2">
            <label className="block text-sm font-medium mb-1">
              Email (optionnel mais utile)
            </label>
            <input
              name="email"
              type="email"
              value={form.email}
              onChange={handleChange}
              className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-primary"
              placeholder="jean@exemple.com"
            />
          </div>

          {/* Source */}
          <div className="md:col-span-2">
            <label className="block text-sm font-medium mb-1">Source du problème *</label>
            <select
              name="source"
              value={form.source}
              onChange={handleChange}
              className={`w-full px-4 py-2 border rounded-lg focus:outline-none focus:ring-2 focus:ring-primary ${
                errors.source ? "border-red-500" : "border-gray-300"
              }`}
            >
              <option value="">-- Choisir --</option>
              <option value="COMPTOIRE">Comptoir</option>
              <option value="CUISINE">Cuisine</option>
              <option value="CHICHA">Chicha</option>
              <option value="EMPLOYE">Personnel/Service</option>
            </select>
            {errors.source && (
              <p className="text-red-500 text-xs mt-1">{errors.source}</p>
            )}
          </div>

          {/* Photo */}
          <div className="md:col-span-2">
            <label className="block text-sm font-medium mb-1">Photo (optionnel)</label>
            <button
              type="button"
              onClick={() => fileInput.current?.click()}
              className="w-full px-4 py-2 border border-dashed border-gray-400 rounded-lg text-gray-600 hover:border-primary hover:text-primary transition"
            >
              📷 Ouvrir la caméra / choisir une image
            </button>
            <input
              ref={fileInput}
              type="file"
              accept="image/*"
              capture="environment"
              name="photo"
              onChange={handleChange}
              className="hidden"
            />
            {form.photo && (
              <p className="text-sm text-gray-600 mt-2">{form.photo.name}</p>
            )}
          </div>

          {/* Message */}
          <div className="md:col-span-2">
            <label className="block text-sm font-medium mb-1">Message *</label>
            <textarea
              name="message"
              value={form.message}
              onChange={handleChange}
              rows={5}
              className={`w-full px-4 py-2 border rounded-lg focus:outline-none focus:ring-2 focus:ring-primary ${
                errors.message ? "border-red-500" : "border-gray-300"
              }`}
              placeholder="Décrivez votre problème..."
            />
            {errors.message && (
              <p className="text-red-500 text-xs mt-1">{errors.message}</p>
            )}
          </div>

          {/* Bouton */}
          <div className="md:col-span-2">
            <button
              type="submit"
              className="w-full bg-primary text-white py-3 rounded-lg shadow hover:bg-orange-600 transition"
            >
              Envoyer la réclamation
            </button>
          </div>
        </form>
      </div>
    </section>
  );
}
